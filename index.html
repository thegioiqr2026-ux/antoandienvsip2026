function lookupName() { const name = document.getElementById('fullName').value.trim(); if (!name) return; document.getElementById('spin').style.display = 'inline-block'; // Tạo lời gọi JSONP const script = document.createElement('script'); const callbackName = 'vdcCallback_' + Math.round(Math.random() * 1000000); window[callbackName] = function(res) { document.getElementById('spin').style.display = 'none'; if (res.exists) { studentInfo = { isUpdate: true, row: res.row }; document.getElementById('foundMsg').style.display = 'block'; document.getElementById('room').value = res.room || ""; document.getElementById('level').value = res.level || ""; } else { studentInfo = { isUpdate: false, row: null }; document.getElementById('foundMsg').style.display = 'none'; if (!confirm("Họ tên này không có trong danh sách. Bạn xác nhận nhập đúng?")) { document.getElementById('fullName').value = ""; } } document.body.removeChild(script); delete window[callbackName]; }; script.src = `${WEB_APP_URL}?action=lookup&fullName=${encodeURIComponent(name)}&callback=${callbackName}`; document.body.appendChild(script); }
